/**
 * @title ChainlinkResultOracle
 * @notice Chainlink Functions consumer that integrates with Oracle to fetch bet results from external APIs.
 * @dev This contract extends FunctionsClient to make Chainlink Functions requests and automatically fulfill
 *      results in the Oracle contract. It handles the request/response cycle for fetching game results.

    event ChainlinkRequestSent(

    /**
     * @notice Constructor for ChainlinkResultOracle.
     * @param router Address of the Chainlink Functions Router.
     * @param initialOwner Owner of the contract.
     * @param _resultOracle Address of the Oracle contract.
     * @param _subscriptionId Chainlink Functions subscription ID.
     * @param _donId DON ID for Chainlink Functions.
     * @param _callbackGasLimit Gas limit for fulfillment callback.
     * @param _sourceCode JavaScript source code for Chainlink Functions.
     */
    constructor(
        address router,
        address initialOwner,
        address _resultOracle,
        uint64 _subscriptionId,
        bytes32 _donId,
        uint32 _callbackGasLimit,
        string memory _sourceCode
    ) FunctionsClient(router) Ownable(initialOwner) {
        if (router == address(0)) revert InvalidResultOracle(router);
        if (_resultOracle == address(0)) revert InvalidResultOracle(_resultOracle);
        if (_subscriptionId == 0) revert InvalidSubscriptionId();
        if (_donId == bytes32(0)) revert InvalidDonId();
        if (_callbackGasLimit == 0) revert InvalidCallbackGasLimit();

        resultOracle = Oracle(_resultOracle);
        subscriptionId = _subscriptionId;
        donId = _donId;
        callbackGasLimit = _callbackGasLimit;
        sourceCode = _sourceCode;
    }

    /**
     * @notice Requests a bet result using Chainlink Functions.
     * @param _game Identifier of the game.
     * @param questionId Identifier of the question.
     * @param params Additional parameters to pass to the API (e.g., API endpoint hints).
     * @return chainlinkRequestId The Chainlink Functions request ID.
     */
    function requestResultWithChainlink(address _game, bytes32 questionId, bytes calldata params)
        external
        returns (bytes32 chainlinkRequestId)
    {
        if (address(resultOracle) == address(0)) revert ResultOracleNotSet();

        // First, create a request in the Oracle to track it
        bytes32 internalRequestId = resultOracle.requestResult(_game, questionId, params);

        // Prepare the Chainlink Functions request
        FunctionsRequest.Request memory req;
        req.initializeRequestForInlineJavaScript(sourceCode);

        // Set args: gameId, betId, optionCount, and option strings
        // args[0] = gameId, args[1] = questionId
        string[] memory args = new string[](1);
        // args[0] = addressToString(_game);
        // args[1] = _uint256ToString(questionId);
        // args[0] = _game.toHexString();
        args[0] = Strings.toHexString(uint256(questionId));

        req.setArgs(args);

        // Encode the request
        bytes memory encodedRequest = req.encodeCBOR();

        // Send the Chainlink Functions request
        chainlinkRequestId = _sendRequest(encodedRequest, subscriptionId, callbackGasLimit, donId);

        // Store the mapping
        _chainlinkRequestToInternalRequest[chainlinkRequestId] = internalRequestId;
        _requestInfo[chainlinkRequestId] = GameBetInfo({game: _game, questionId: questionId, requester: msg.sender});

        emit ChainlinkRequestSent(chainlinkRequestId, internalRequestId, _game, questionId);
    }

    /**
     * @notice Callback function called by Chainlink Functions when the request is fulfilled.
     * @param requestId The Chainlink Functions request ID.
     * @param response The response from the Chainlink Functions execution.
     * @param err Error bytes if the request failed.
     */
    function _fulfillRequest(bytes32 requestId, bytes memory response, bytes memory err) internal override {
        bytes32 internalRequestId = _chainlinkRequestToInternalRequest[requestId];
        if (internalRequestId == bytes32(0)) {
            revert RequestNotFound(requestId);
        }

        if (err.length > 0) {
            // Request failed, mark as failed in Oracle
            string memory reason = string(err);
            resultOracle.failRequest(internalRequestId, reason);
            emit ChainlinkRequestFailed(requestId, internalRequestId, reason);
        } else {
            // Parse the response to get the option index (uint256)
            // uint256 resultIndex = _parseResultIndex(response);
            uint256[] memory result = abi.decode(response, (uint256[]));

            // Validate that the index is within bounds
            GameBetInfo memory info = _requestInfo[requestId];
            // _validateOptionIndex(resultIndex, info.game, info.betId);

            // Fulfill the result in Oracle
            resultOracle.requestResult(info.game, info.questionId, response);
            resultOracle.fulfillResult(info.questionId, result, response);

            // Get the option key for the event
            // bytes32[] memory optionKeys = game.bets[info.betId].;
            // bytes32 resultOption = optionKeys[resultIndex];
            emit ChainlinkRequestFulfilled(requestId, internalRequestId, result);
        }

        // Clean up mappings
        delete _chainlinkRequestToInternalRequest[requestId];
        delete _requestInfo[requestId];
    }

    /**
     * @notice Parses the option index (uint256) from the Chainlink Functions response.
     * @param response The response bytes from Chainlink Functions.
     * @dev The response is expected to be a uint256 encoded using Functions.encodeUint256().
     * @return resultIndex The parsed option index.
     */
    function _parseResultIndex(bytes memory response) internal pure returns (uint256 resultIndex) {
        if (response.length == 0) {
            revert("Empty response from Chainlink Functions");
        }
        // Decode the uint256 directly from the response
        resultIndex = abi.decode(response, (uint256));
    }

     * @notice Updates the subscription ID.

    /**
     * @notice Updates the Oracle contract address.
     * @param newResultOracle The new Oracle address.
     */
    function updateResultOracle(address newResultOracle) external onlyOwner {
        if (newResultOracle == address(0)) revert InvalidResultOracle(newResultOracle);
        resultOracle = Oracle(newResultOracle);
    }

    /**
     * @notice Gets the internal request ID for a Chainlink request ID.
     * @param chainlinkRequestId The Chainlink Functions request ID.

        return _chainlinkRequestToInternalRequest[chainlinkRequestId];
